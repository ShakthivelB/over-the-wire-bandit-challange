#!/bin/bash

Myname=$(whoami)
Mytarget=$(echo I am user $myname | md5sum | cut -d ' ' -f 1)

Echo "Copying passwordfile /etc/bandit_pass/$myname to /tmp/$mytarget"

Cat /etc/bandit_pass/$myname > /tmp/$mytarget
